Gradebook Not Updated - Student Report

~[tlist_sql;SELECT DISTINCT S.LASTFIRST, T.LASTFIRST, C.COURSE_NAME, CC.SECTION_NUMBER, CASE WHEN CC.LASTGRADEUPDATE < (SELECT TRX.FIRSTDAY FROM TERMS TRX WHERE TRX.SCHOOLID = CC.SCHOOLID AND TRX.FIRSTDAY < = SYSDATE AND TRX.LASTDAY >= SYSDATE AND TRX.PORTION = 1) THEN 'Never' ELSE TO_CHAR(CC.LASTGRADEUPDATE) END "LASTGRADEUPDATE" FROM CC, TEACHERS T, COURSES C, TERMS TR, STUDENTS S WHERE T.ID = CC.TEACHERID AND C.COURSE_NUMBER = CC.COURSE_NUMBER AND TR.ID = CC.TERMID AND S.ID = CC.STUDENTID AND TR.SCHOOLID = CC.SCHOOLID AND CC.TERMID IN (SELECT TRX.ID FROM TERMS TRX WHERE TRX.SCHOOLID = CC.SCHOOLID AND TRX.FIRSTDAY < = SYSDATE AND TRX.LASTDAY >= SYSDATE) AND CC.LASTGRADEUPDATE < SYSDATE-8 AND CC.SCHOOLID = ~(CURSCHOOLID) AND C.COURSE_NAME NOT LIKE 'Advisory%' AND C.COURSE_NAME NOT LIKE 'Guided%' AND C.COURSE_NUMBER NOT LIKE 'STUDY' AND C.COURSE_NUMBER NOT LIKE 'ELEMAM' AND C.COURSE_NUMBER NOT LIKE 'ELEMPM' AND C.COURSE_NAME NOT LIKE 'Behavior%' AND C.COURSE_NUMBER NOT LIKE 'caseload' AND sec.excludefromstoredgrades != 1 AND (c.excludefromstoredgrades = 0 AND sec.excludefromstoredgrades IN (0,2)) AND S.ENTRYDATE < SYSDATE-15 ORDER BY 2,3 ;alternatecolor] [/tlist_sql]
