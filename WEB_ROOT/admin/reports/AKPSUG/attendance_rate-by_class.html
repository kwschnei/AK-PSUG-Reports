Attendance Rate by Class

~[tlist_sql; WITH termdates AS ( SELECT cc.id, cc.dateenrolled, CASE WHEN cc.dateleft>SYSDATE THEN SYSDATE ELSE cc.dateleft END AS dateleft FROM cc WHERE ABS(cc.termid) IN (SELECT tr.id FROM terms tr WHERE tr.yearid=~(curyearid)) AND cc.schoolid=~(curschoolid) AND cc.dateenrolled=termdates.dateenrolled AND cd.schoolid=cc.schoolid AND cd.membershipvalue!=0) AS daysinclass FROM cc INNER JOIN termdates ON termdates.id=cc.id JOIN sections sec ON sec.id=ABS(cc.sectionid) WHERE NVL(sec.exclude_ada,0)!=1 ), section_data AS ( SELECT sccd.sectionid, SUM(sccd.currentabsences) AS absences, SUM(sccd.daysinclass) AS daysinclass FROM ccdata sccd GROUP BY sccd.sectionid ) SELECT DISTINCT tr.name, cr.course_name, sec.expression, u.lastfirst, ROUND(((sd.daysinclass-sd.absences)/(sd.daysinclass)*100),2)||'%' FROM section_data sd JOIN sections sec ON sec.id=sd.sectionid JOIN courses cr ON cr.course_number=sec.course_number JOIN schoolstaff ss ON ss.id=sec.teacher JOIN users u ON u.dcid=ss.users_dcid JOIN terms tr ON tr.id=sec.termid WHERE cr.exclude_ada!=1 ORDER BY tr.name,u.lastfirst,sec.expression,cr.course_name ;alternatecolor] [/tlist_sql]
