Tardies by Class

~[tlist_sql; SELECT sch.abbreviation "School", t.lastfirst "Teacher", c.COURSE_NAME "Class", cc.expression "Period", sec.no_of_students "# of Students", count (*) "Tardies" FROM students s JOIN schools sch ON sch.school_number = s.schoolid JOIN attendance att ON att.studentid = s.id JOIN attendance_code ac ON ac.id = att.attendance_codeid AND (ac.att_code in ('T','TU','TEX','TUN') OR UPPER(ac.description) LIKE '%TARDY%') JOIN cc ON cc.id = att.CCID AND cc.termid in (SELECT tr.id FROM terms tr WHERE tr.schoolid = s.schoolid AND tr.firstday <= SYSDATE AND tr.lastday >= SYSDATE) JOIN courses c ON c.course_number = cc.course_number JOIN teachers t ON t.id = cc.teacherid JOIN sections sec ON cc.sectionid = sec.id WHERE s.enroll_status = 0 AND s.schoolid = ~(curschoolid) GROUP BY sch.abbreviation, t.lastfirst, c.COURSE_NAME, cc.expression, sec.no_of_students ORDER BY 6 DESC ;alternatecolor] [/tlist_sql]
